@Component({
  selector: "bo-drilldown",
  standalone: true,
  imports: [
    PoLoadingModule,
    PoTableModule,
    PoBreadcrumbModule,
    CommonModule,
    ExportButtonComponent,
  ],
  templateUrl: "./drilldown.component.html",
  styleUrl: "./drilldown.component.css",
})


export class DrilldownComponent {
  /**
   * Nome do Drilldown, vai aparecer na raiz do breadcrumbs.
   * @example
   * 'Faturamento'
   */
  @Input() drillDownName: string = "";

  /**
   * Ativa ou desativa botão de exportar dados.
   * @example
   * false
   */
  @Input() exportButton: boolean = false;

  /**
   * Escolhe se quer exportar em pdf, excel ou ambos..
   * @example
   * pdf;excel
   */
  @Input() formatExport: string = "";

  /**
   * Altura/Tamanho da tabela.
   * @example
   * 400
   */
  @Input() drillDownHeight: number = 400;

  /**
   * Dados do Drilldown
   * @example
   * {  header: any[];  items: any[];  hasNext: boolean;  }
   */
  @Input() drillDownData: DrillDownData = {
    header: [""],
    items: [],
    hasNext: false,
  };

  /**
   * Todos os dados para export
   * @example
   * {  header: any[];  items: any[];  hasNext: boolean;  }
   */
  @Input() allDrillDownData: DrillDownData = {
    header: [""],
    items: [],
    hasNext: false,
  };

  /**
   * Evento de clique em um action link.
   */
  @Output() actionLinkClicked: EventEmitter<any> = new EventEmitter<any>();

  /**
   * Evento de clique em carregar mais itens.
   */
  @Output() loadMoreClicked: EventEmitter<any> = new EventEmitter<any>();

  /**
   * Evento de clique em carregar mais itens.
   */
  @Output() getAllItemsToExport: EventEmitter<any> = new EventEmitter<any>();

  /**
   *  Flag para verificar se o drilldown foi iniciado.
   */
  drillInitiated: boolean = false;

  /**
   * Colunas da tabela.
   * @example
   * [{ "property": "order", "label": "Pedido", "type": "link", "action": "Link", "showFilter": true, "visible": true }]
   */
  drillColumns: Array<PoTableColumn> = [];

  /**
   *  Itens da tabela.
   * @example
   * [{ "order": "A00812", "item": "01", "sequen": "01", "client": "005024", "store": "00", "name": "5M COMERCIO ATACADIS", "product": "00040065901", "prodname": "ACHOC LIQ ITALAC 1L C/ TAMPA", "prodgroup": "", "saleprice": 3.43, "sale": 480, "amount": 1646.4, "lot": "", "sublot": "", "deliverydate": "15/02/2022", "emission": "15/02/2022", "paymentterms": "003", "amountsc6": "", "cfop": "6603", "serialnum": "" }]
   */
  drillItems: Array<any> = []; // Itens da tabela.

  /**
   *  Níveis de breadcrumbs da bandeja.
   * @example
   * [ { "label": "Faturamento" }, { "label": "A00812" }, { "label": "005024" } ]
   */
  drillBreadcrumbs: PoBreadcrumbItem[] = [];

  /**
   * Flag para verificar se o drilldown está carregando.
   */
  drillIsLoading: boolean = false;

  constructor(private changeDetector: ChangeDetectorRef) {}

  /**
   * Lifecycle hook que é chamado quando qualquer propriedade vinculada a dados de uma diretiva é alterada.
   * @param changes - mudanças.
   * @autor Matheus Camargo B. da Costa
   */
  ngOnChanges(changes: SimpleChanges): void {
    if (changes["drillDownData"]) {
      const previousValue = changes["drillDownData"].previousValue;
      const currentValue = changes["drillDownData"].currentValue;

      if (JSON.stringify(previousValue) !== JSON.stringify(currentValue)) {
        this.updateComponent();
      } else {
        console.log("No changes detected in data");
        this.drillIsLoading = false;
      }
    }

    if (changes["allDrillDownData"]) {
      const previousValue = changes["allDrillDownData"].previousValue;
      const currentValue = changes["allDrillDownData"].currentValue;

      if (JSON.stringify(previousValue) !== JSON.stringify(currentValue)) {
        this.drillIsLoading = false;
      } else {
        this.drillIsLoading = false;
      }
    }

    if (!this.drillInitiated) {
      this.addBreadcrumb();
      this.drillInitiated = true;
    }
  }

  /**
   * Atualiza o componente com novos dados.
   * @autor Matheus Camargo B. da Costa
   */
  updateComponent(): void {
    if (
      this.drillDownData.header.length === 0 ||
      this.drillDownData.items.length === 0
    )
      return;
    this.drillColumns = this.drillDownData.header;
    this.drillItems = this.drillDownData.items;
    this.drillIsLoading = false;
  }

  /**
   * Navega um nivel para baixo na hierarquia de drilldown.
   * @param row - Row selecionada.
   * @param clickedValue - Dado do link que clicou.
   * @autor Matheus Camargo B. da Costa
   */
  navigateDown(row: any, clickedValue: string): void {
    if (!row || !clickedValue) {
      console.error("Erro: Propriedade inexistente ou valor inválido");
      return;
    }
    this.addBreadcrumb(row, clickedValue);
    this.drillIsLoading = true;
    this.actionLinkClicked.emit({ row, clickedValue });
  }

  /**
   * Navega um nivel para cima na hierarquia de drilldown.
   * @param row - Row selecionada.
   * @param clickedValue - Dado do link que clicou.
   * @autor Matheus Camargo B. da Costa
   */
  navigateUp(row: any, clickedValue: any): void {
    this.getBackToBreadcrumb(clickedValue);
    this.drillIsLoading = true;
    this.actionLinkClicked.emit(row);
  }

  /**
   * Carrega mais itens da lista.
   * @autor Matheus Camargo B. da Costa
   */
  loadMoreItems(): void {
    this.drillIsLoading = true;
    this.loadMoreClicked.emit();
  }

  /**
   * Adiciona um breadcrumb para a bandeja de breadcrumbs.
   * @param row - Row selecionada (opcional).
   * @param clickedValue - Valor clicado (opcional).
   * @autor Matheus Camargo B. da Costa
   */
  addBreadcrumb(row: any = null, clickedValue: string = ""): void {
    const breadcrumbItem: PoBreadcrumbItem = {
      label: row == null ? this.drillDownName : clickedValue,
      action: () => {
        if (row) this.navigateUp(row, clickedValue);
        else this.getBackToRoot(this.drillDownName);
      },
    };

    for (let breacrumb of this.drillBreadcrumbs) {
      if (breacrumb.label === breadcrumbItem.label) {
        return;
      }
    }

    this.drillBreadcrumbs.push(breadcrumbItem);
    this.changeDetector.detectChanges();
  }

  /**
   * Navega de volta para uma breadcrumb especifica.
   * @param clickedValue - Valor clicado.
   * @autor Matheus Camargo B. da Costa
   */
  getBackToBreadcrumb(clickedValue: any): void {
    const index = this.drillBreadcrumbs.findIndex(
      (breadcrumb) => breadcrumb.label === clickedValue
    );
    if (index !== -1) {
      this.drillBreadcrumbs = this.drillBreadcrumbs.slice(0, index + 1);
    }
  }

  /**
   * Navega de volta para raiz da tabela.
   * @param clickedValue - Valor clicado.
   * @autor Matheus Camargo B. da Costa
   */
  getBackToRoot(clickedValue: any): void {
    const index = this.drillBreadcrumbs.findIndex(
      (breadcrumb) => breadcrumb.label === clickedValue
    );
    if (index !== -1) {
      this.drillBreadcrumbs = this.drillBreadcrumbs.slice(0, index + 1);
    }
    this.drillIsLoading = true;
    this.actionLinkClicked.emit(null);
  }

  getAllItems(): void {
    this.getAllItemsToExport.emit();
  }
}

export interface DrillDownData {
  header: any[];
  items: any[];
  hasNext: boolean;
}

<div class="wrapper">

        </div>
        <ng-template #noHyperlink>
          {{ row[column.property] }}
        </ng-template>
      </ng-template>
    </po-table>
  </div>
</div>
