import { Chart, ChartConfiguration, ChartType, registerables } from "chart.js";
import {
  Component,
  Input,
  AfterViewInit,
  ElementRef,
  ViewChild,
} from "@angular/core";

@Component({
  selector: "bo-charts",
  standalone: true,
  imports: [],
  templateUrl: "./charts.component.html",
  styleUrls: ["./charts.component.css"],
})
export class ChartsComponent implements AfterViewInit {
  /**
   * Largura do gráfico em pixels.
   * @example
   * 300
   */
  @Input() width: number = 300;

  /**
   * Altura do gráfico em pixels.
   * @example
   * 300
   */
  @Input() height: number = 300;

  /**
   * Tipo do gráfico a ser renderizado.
   * @example
   * 'bar' | 'line' | 'doughnut' | 'radar' | 'polarArea' | 'scatter'
   */
  @Input() type: ChartType = "bar";

  /**
   * Rótulos dos eixos X ou categorias do gráfico.
   * @example
   * ['Janeiro', 'Fevereiro', 'Março']
   */
  @Input() labels: string[] = [];

  /**
   * Dados do gráfico, incluindo valores, cores e estilos.
   * @example
   * [
   *   { label: 'Dataset 1', data: [10, 20, 30], backgroundColor: ['red', 'blue', 'yellow'] }
   * ]
   */
  @Input() datasets: any[] = [];

  /**
   * Elemento HTML do canvas onde o gráfico será renderizado.
   */
  @ViewChild("chartCanvas") chartCanvas!: ElementRef<HTMLCanvasElement>;

  /**
   * Método chamado após a inicialização da view.
   * Configura e renderiza o gráfico utilizando a biblioteca Chart.js.
   */
  ngAfterViewInit(): void {
    Chart.register(...registerables);

    /**
     * Configuração do gráfico, incluindo tipo, dados e opções.
     */
    const config: ChartConfiguration = {
      type: this.type,
      data: {
        labels: this.labels,
        datasets: this.datasets,
      },
      options: {
        responsive: true,
        maintainAspectRatio: false,
      },
    };

    /**
     * Obtém o contexto 2D do canvas e cria o gráfico.
     * Exibe um erro no console se o contexto não for encontrado.
     */
    const ctx = this.chartCanvas.nativeElement.getContext("2d");
    if (ctx) {
      new Chart(ctx, config);
    } else {
      console.error("Erro: Não foi possível obter o contexto do canvas.");
    }
  }
}
