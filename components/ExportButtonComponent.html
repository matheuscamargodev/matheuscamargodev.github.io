  /**

  exportAllAction: PoModalAction = {

  /**
   * AÃ§Ãµes a serem exibidas na tela, vocÃª pode optar por pdf, excel ou ambas separadas por ponto e virgula para exportaÃ§Ã£o.

  @Input() dataToExport: any = [];

  @Input() allDataToExport: any = [];

  constructor(private changeDetector: ChangeDetectorRef) {}

  /**
   * Lifecycle hook que Ã© chamado quando qualquer propriedade vinculada a dados de uma diretiva Ã© alterada.
   * @param changes - mudanÃ§as.
   * @autor Matheus Camargo B. da Costa
   */
  ngOnChanges(changes: SimpleChanges): void {
    const propertiesToCheck = ["actions", "allDataToExport"];

    propertiesToCheck.forEach((property) => {
      if (changes[property]) {
        const previousValue = changes[property].previousValue;
        const currentValue = changes[property].currentValue;

        if (JSON.stringify(previousValue) !== JSON.stringify(currentValue)) {
          if (property === "actions") {
            this.updateComponent();
          } else if (property === "allDataToExport") {
            if (this.currentExporting === "excel") {
              this.exportExcellAllData();
            } else {
              this.exportPDFAllData();
            }
          }
        } else {
          console.log(`No changes detected in ${property}`);
        }
      }
    });
  }

  /**
   * Atualiza as aÃ§Ãµes do componente de acordo com as aÃ§Ãµes disponÃ­veis.
   * @autor Matheus Camargo B. da Costa
   */
  updateComponent(): void {
    if (this.actions != "") {
      this.buttonActions = [];
      for (let format of this.actions.split(";")) {
        if (format == "pdf") {
          this.buttonActions.push({
            label: this.customTextExportPDF,
            action: this.exportPDFOpenModal.bind(this),
          });
        }
        if (format == "excel") {
          this.buttonActions.push({
            label: this.customTextExportExcel,
            action: this.exportExcelOpenModal.bind(this),
          });
        }
      }
      this.changeDetector.detectChanges();
    }
  }

  /**
   * Verifica o tipo de exportaÃ§Ã£o e chama a funÃ§Ã£o correspondente.
   * @param type - Tipo de exportaÃ§Ã£o ('current' ou 'all').
   * @autor Matheus Camargo B. da Costa
   */
  checkExportation(type: string): void {
    const exportActions: any = {
      excel: {
        current: this.exportExcelCurrent.bind(this),
        all: this.exportExcelAll.bind(this),
      },
      pdf: {
        current: this.exportPDFCurrent.bind(this),
        all: this.exportPDFAll.bind(this),
      },
    };

    const exportType = this.currentExporting === "excel" ? "excel" : "pdf";
    const action = exportActions[exportType][type];

    if (action) {
      action();
    } else {
      console.error(`Invalid export type: ${type}`);
    }
  }

  /**

  exportExcellAllData(): void {
    if (!this.allDataToExport || this.allDataToExport.length === 0) {
      console.error('No data available to export to Excel');
      return;
    }
    this.saveFileExcel(this.allDataToExport, "export");
    this.closeModal();
  }

  /**
   * Exporta os dados atuais para Excel.
   * @autor Matheus Camargo B. da Costa
   */
  exportExcelCurrent(): void {
    if (!this.dataToExport || this.dataToExport.length === 0) {
      console.error('No data available to export to Excel');
      return;
    }
    this.saveFileExcel(this.dataToExport, "export");
    this.closeModal();
  }

  /**
   * Salva os dados em um arquivo Excel.
   * @param data - Dados a serem exportados.
   * @param fileName - Nome do arquivo.
   * @autor Matheus Camargo B. da Costa
   */
  saveFileExcel(data: any[] | null, fileName: string): void {
    if (!data || data.length === 0) {
      console.error('No data available to export to Excel');
      return;
    }
    const worksheet: XLSX.WorkSheet = XLSX.utils.json_to_sheet(data);
    const workbook: XLSX.WorkBook = XLSX.utils.book_new();
    XLSX.utils.book_append_sheet(workbook, worksheet, "Sheet1");
    const excelBuffer: any = XLSX.write(workbook, {
      bookType: "xlsx",
      type: "array",
    });
    const blob: Blob = new Blob([excelBuffer], {
      type: "application/octet-stream",
    });
    saveAs(blob, `${fileName}.xlsx`);
  }

  /**
   * Abre o modal de exportaÃ§Ã£o para PDF.
   * @autor Matheus Camargo B. da Costa
   */
  exportPDFOpenModal(): void {
    this.currentExporting = "pdf";
    this.exportModal.open();
  }

  /**
   * Exporta todos os itens para PDF.
   * @autor Matheus Camargo B. da Costa
   */
  exportPDFAll(): void {
    this.getAllItemsToExport.emit();
    this.closeModal();
  }

  /**
   * Exporta os dados atuais para PDF.
   * @autor Matheus Camargo B. da Costa
   */
  exportPDFCurrent(): void {
    if (!this.dataToExport || this.dataToExport.length === 0) {
      console.error('No data available to export to Excel');
      return;
    }
    this.saveFilePDF(this.dataToExport, "export");
    this.closeModal();
    setTimeout(() => {
      this.allDataToExport = [];
    }, 2000);
  }

  /**
   * Exporta todos os dados para PDF.
   * @autor Matheus Camargo B. da Costa
   */
  exportPDFAllData(): void {
    if (!this.allDataToExport || this.allDataToExport.length === 0) {
      console.error('No data available to export to Excel');
      return;
    }
    this.saveFilePDF(this.allDataToExport, "export");
    this.closeModal();
    setTimeout(() => {
      this.allDataToExport = [];
    }, 2000);
  }

  /**
   * Salva os dados em um arquivo PDF.
   * @param data - Dados a serem exportados.
   * @param fileName - Nome do arquivo.
   * @autor Matheus Camargo B. da Costa
   */
  saveFilePDF(data: any[] | null, fileName: string): void {
    if(!data) return;
    const filteredData: any = data.map((item) => ({
      Pedido: item.order,
      Item: item.item,
      Sequencia: item.sequen,
      Cliente: item.client,
      Loja: item.store,
      NomeCliente: item.name,
      Produto: item.product,
      DescricaoProduto: item.prodname,
      PrecoVenda: item.saleprice,
      Quantidade: item.sale,
      Total: item.amount,
    }));
    const doc = new jsPDF({ format: "a3", orientation: "landscape" });
    const headers = Object.keys(filteredData[0]);
    const rows = filteredData.map((obj: any) => headers.map((key) => obj[key]));

    autoTable(doc, {
      head: [headers],
      body: rows,
      columnStyles: {
        0: { cellWidth: 25 },
        1: { cellWidth: 25 },
        2: { cellWidth: 25 },
        3: { cellWidth: 25 },
        4: { cellWidth: 25 },
        5: { cellWidth: 80 },
        6: { cellWidth: 40 },
        7: { cellWidth: 80 },
        8: { cellWidth: 20 },
        9: { cellWidth: 20 },
      },
      styles: { fontSize: 10 },
    });

    doc.save(`${fileName}.pdf`);
  }
}
